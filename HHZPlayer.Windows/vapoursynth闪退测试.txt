第三步：明确指定模型路径
import vapoursynth as vs
core = vs.core
core.std.LoadPlugin(r"rife\librife_windows_x86-64.dll")

model = r"rife\models\rife-v4.6"
clip = core.std.BlankClip(width=1280, height=720, format=vs.RGBS)
clip = core.rife.RIFE(clip, model_path=model, factor_num=2, gpu_id=0)
clip.set_output()
